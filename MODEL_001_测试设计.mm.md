│   │   │   └── 极高薪：1000000（最大允许值）
│   │   └── 无效等价类
│   │       ├── 0 或负数
│   │       └── 超过最大范围
│   └── 1.4 social_security_flag（社保标志）
│       ├── 有效等价类
│       │   ├── 有社保 = 1
│       │   └── 无社保 = 0
│       └── 无效等价类
│           ├── 其他值（2、-1 等）

├── 2. 边界值分析
│   ├── 2.1 月薪边界
│   │   ├── monthly_salary = 10000（临界值，应不准入）
│   │   ├── monthly_salary = 10001（刚过门槛，应准入）
│   │   ├── monthly_salary = 9999（刚不及格，应不准入）
│   │   ├── monthly_salary = 1（最小合法值，不准入）
│   │   └── monthly_salary = 10000000（最大合法值）
│   ├── 2.2 余额边界

│       ├── user_level = 1（最低等级）
│       └── user_level = 5（最高等级）

│   ├── 3.1 准入条件组合（2 个条件 AND）
│   │   ├── 满足准入条件
│   │   │   ├── 工资>10000 ✓ + 有社保 ✓ → admit_flag=1
│   │   │   └── 子分支：正资产（coefficient=2.3）
│   │   │   └── 子分支：零资产（coefficient=0.5）

│   │   └── 不满足准入条件
│   │       ├── 工资≤10000 ✓ + 有社保 ✓ → admit_flag=0
│   │       ├── 工资>10000 ✓ + 无社保 ✓ → admit_flag=0
│   │       └── 工资≤10000 ✓ + 无社保 ✓ → admit_flag=0
│   └── 3.2 系数取值逻辑
│       ├── admit_flag=1 ∧ balance>0 → coefficient=2.3
│       ├── admit_flag=1 ∧ balance=0 → coefficient=0.5
│       ├── admit_flag=1 ∧ balance<0 → coefficient=0.5
│       └── admit_flag=0 → coefficient=0
│
├── 4. 额度计算验证
│   ├── 4.1 正常计算公式验证
│   │   ├── credit_limit = user_level × (balance÷1000) × salary × coefficient
│   │   └── 精度验证：两位小数截断/四舍五入
│   ├── 4.2 负值归零处理
│   │   ├── 原始计算为负时强制归零
│   │   └── 例如：1×(-5)×15000×0.5 = -37500 → 0.00
│   └── 4.3 不准入情况
│       ├── admit_flag=0 时强制 credit_limit=0.00
│       └── 系数强制为 0
│
├── 5. 异常流程
│   ├── 5.1 用户不存在

├── 6. 风险区域
│   ├── 6.1 负资产场景

│   │   ├── 重度负资产（如 -100000）
│   │   └── 确保所有负资产都正确归零
│   ├── 6.2 临界值精度
│   │   ├── 工资=10000.00 vs 10000.01
│   │   └── 余额=0.00 vs 0.01 vs -0.01
│   └── 6.3 大数乘法溢出
